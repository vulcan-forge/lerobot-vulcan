```
WORKDIR="/home/sourccey/.cache/huggingface/lerobot/Combination/WorkingDirectory"
OUT_REPO="Combination/workingdirectory_all_$(date +%Y%m%d_%H%M%S)"
OUT_ROOT="/home/sourccey/.cache/huggingface/lerobot/${OUT_REPO}"
CFG="/tmp/merge_workingdirectory.json"

export WORKDIR OUT_REPO OUT_ROOT CFG

uv run python - <<'PY'
import json, os
from pathlib import Path

workdir = Path(os.environ["WORKDIR"])
out_repo = os.environ["OUT_REPO"]
out_root = Path(os.environ["OUT_ROOT"])
cfg_path = Path(os.environ["CFG"])

roots = sorted({p.parent.parent for p in workdir.rglob("meta/info.json")})
if len(roots) < 2:
    raise SystemExit(f"Need at least 2 datasets, found {len(roots)}")

repo_ids = [f"local/{r.relative_to(workdir).as_posix().replace('/','__')}" for r in roots]
cfg = {
    "new_repo_id": out_repo,
    "new_root": str(out_root),
    "operation": {
        "type": "merge",
        "repo_ids": repo_ids,
        "roots": [str(r) for r in roots],
    },
}
cfg_path.write_text(json.dumps(cfg, indent=2))
print(f"Wrote {cfg_path} with {len(roots)} datasets")
PY

uv run lerobot-edit-dataset --config_path "$CFG"
```
