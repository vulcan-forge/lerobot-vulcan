This is an MD file for implement the stage aware reward model for datasets

###

Training the model

```
CUDA_VISIBLE_DEVICES=0 uv run lerobot-train \
 --dataset.repo_id="Combination/sourccey-shirt-fold-c-005" \
 --reward_model.type=sarm \
 --reward_model.annotation_mode=single_stage \
 --reward_model.image_keys='["observation.images.front_right","observation.images.front_left"]' \
 --output_dir=outputs/train/sarm_sourccey-shirt-fold-c-005 \
 --batch_size=8 \
 --steps=25000 \
 --save_freq=5000 \
 --wandb.enable=false \
 --reward_model.push_to_hub=false
```

### Resume Training

```
CUDA_VISIBLE_DEVICES=0 uv run lerobot-train \
  --resume=true \
  --config_path=outputs/train/sarm_sourccey-shirt-fold-c-005/checkpoints/last/pretrained_model/train_config.json

```

Compute the RABC Weights

###

Visualize Dryrun

```
uv run python -m lerobot.rewards.sarm.compute_rabc_weights \
 --dataset-repo-id="Combination/sourccey-shirt-fold-c-005" \
 --reward-model-path="outputs/train/sarm_sourccey-shirt-fold-c-005/checkpoints/025000/pretrained_model" \
 --head-mode=sparse \
 --device=cuda \
 --output-dir="./outputs/sarm_viz" \
 --stride=30 \
 --visualize-only \
 --visualize-episode-indices=32

```

Visualize Multiple Ranked

```
uv run python -m lerobot.rewards.sarm.compute_rabc_weights \
 --dataset-repo-id="Combination/sourccey-shirt-fold-c-005" \
 --reward-model-path="outputs/train/sarm_sourccey-shirt-fold-c-005/checkpoints/025000/pretrained_model" \
 --head-mode=sparse \
 --device=cuda \
 --output-dir="./outputs/sarm_viz_ranked" \
 --stride=30 \
 --visualize-only \
 --visualize-ranking-csv="outputs/sarm_episode_ranking_progress/episode_ranking_progress.csv" \
 --visualize-top-k=25 \
 --visualize-bottom-k=25
```

Generate RA-BC progress

```
uv run python -m lerobot.rewards.sarm.compute_rabc_weights \
 --dataset-repo-id="Combination/sourccey-shirt-fold-c-005" \
 --reward-model-path="outputs/train/sarm_sourccey-shirt-fold-c-005/checkpoints/025000/pretrained_model" \
 --head-mode=sparse \
 --device=cuda \
 --num-visualizations=10 \
 --output-dir="./outputs/sarm_viz_weights" \
 --stride=30 \
 --no-push-to-hub
```
